newtype Parser a

The type of parsers.

Each parser is seen as a function taking an input string to a pair between the parse result and the rest of the input, if the parsing succeeds, or to Nothing, if the parsing fails. The function type is wrapped into a newtype declaration, for implementation hiding. The parser type is parametrized over the type of its result, a.

failure :: Parser a

Never parses anything, always returning Nothing.

success :: a -> Parser a

Parses an explicitly given value, without consuming any input.

Examples: 

>>> getParser (success 1) "abc"
Just (1, "abc")

token :: Char -> Parser Char

Parses a given character.

Examples: 

>>> getParser (token 'a') "abc"
Just ('a', "bc")


>>> getParser (token 'a') "bbc"
Nothing

spot :: (Char -> Bool) -> Parser Char

Parses a character that satisfies a given property.

Examples: 

>>> getParser (spot isLetter) "abc"
Just ('a', "bc")


>>> getParser (spot isLetter) "123"
Nothing

letter :: Parser Char

Parses a letter at the beginning of the input string.

Examples: 

>>> getParser letter "abc"
Just ('a', "bc")


>>> getParser letter "123"
Nothing

digit :: Parser Int

Parses a digit at the beginning of the input string, and returns the result as an Int:

Examples: 

>>> getParser digit "123"
Just (1, "23")


>>> getParser digit "abc"
Nothing

digitLetter :: Parser (Char, Int)

Parses a letter followed by a digit.

Examples: 

>>> getParser digitLetter "a12"
Just (('a', 1), "2")

operation :: Parser (Int, Char, Int)

Parses an expression of the form <digit> <operator> <digit>, where <operator> can be any of +, -, *, /.

Examples: 

>>> getParser operation "1+2"
Just ((1, '+', 2), "")

insensitiveA :: Parser Char

Parses an 'A' in a case insensitive manner.

Examples: 

>>> getParser insensitiveA "ab"
Just ('a', "b")


>>> getParser insensitiveA "Ab"
Just ('A', "b")

insensitiveAs :: Parser String

Parses 1 or more 'A's in a case insensitive manner.

Examples: 

>>> getParser insensitiveAs "aAab"
Just ("aAa", "b")
